import sys

import pytest

sys.path.insert(0, r"C:\Users\ys_tk\claude")

@pytest.fixture(scope="module")

    widget.insert("1.0", text)
    app = object.__new__(tc.ThreadCuttingApp)
    app._C = tc.ThreadCuttingApp._C
    tc.ThreadCuttingApp._apply_syntax_highlight(app, widget)

    return widget.tag_names(index)


def test_comment_tagged(widget):

    assert "comment" in tags_at(widget, "1.2")

    assert "axis" in tags_at(widget, "1.15")     # Z

`ThreadCuttingApp` クラスに以下2メソッドを追加（`_set_text` の直前あたり）：

```python
    def _configure_highlight_tags(self, widget: tk.Text) -> None:
        """Syntax highlight タグを widget に設定する（優先度: 後に定義 = 高優先）。"""
        # num → axis → gm_code → n_line → o_pct → comment の順で定義
        # comment が最高優先になり、コメント内の他タグを上書きする
        widget.tag_configure("num",     foreground="#a6e3a1")
        widget.tag_configure("axis",    foreground="#f9e2af")
        widget.tag_configure("gm_code", foreground="#4a9fd4")
        widget.tag_configure("n_line",  foreground="#cba6f7")
        widget.tag_configure("o_pct",   foreground="#f38ba8")
        widget.tag_configure("comment", foreground="#6c7086",
                             font=("Consolas", 10, "italic"))

    def _apply_syntax_highlight(self, widget: tk.Text) -> None:
        """widget 内のテキスト全体に G コード syntax highlight タグを適用する。"""
        import re

        # 既存タグをすべてクリア
        for tag in ("num", "axis", "gm_code", "n_line", "o_pct", "comment"):
            widget.tag_remove(tag, "1.0", "end")

        text = widget.get("1.0", "end")

        patterns = [
            ("num",     r'-?\b\d+\.?\d*'),
            ("axis",    r'[XZIFPUHDBRQ](?=[\-\d])'),
            ("gm_code", r'\b[GM]\d+(?:\.\d+)?'),
            ("n_line",  r'\bN\d+'),
            ("o_pct",   r'(?m)^[Oo]\d+|^%'),
            ("comment", r'\(.*?\)'),
        ]

        for tag, pattern in patterns:
            for m in re.finditer(pattern, text):
                start = f"1.0 + {m.start()} chars"
                end   = f"1.0 + {m.end()} chars"
                widget.tag_add(tag, start, end)
```
